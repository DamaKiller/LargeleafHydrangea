# 输出每个月的日期
```
year = 2023
month = 3
start_day = datetime(year, month, 1)
end_day = datetime(year, month, calendar.monthrange(year, month)[1], 23, 59, 59)
date_list = []
while start_day <= end_day:
    date_list.append(datetime.strftime(start_day, '%Y-%m-%d'))
    start_day += timedelta(days=1)
```
